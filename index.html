// ================================================================ // http-functions.js v8 // getJobs: limit 200→1000、detailフィールドを省略して軽量化 // ================================================================ import { ok, serverError } from "wix-http-functions"; import wixData from "wix-data"; var COLLECTION = "Jobs"; // ──────────────────────────────────────────── // 1) チャット用：求人データ取得（GET） // ★ 1000件返却・detailフィールドは省略して高速化 // ──────────────────────────────────────────── export function get_getJobs(request) { var headers = { "Access-Control-Allow-Origin": "*", "Content-Type": "application/json; charset=utf-8" }; return wixData.query(COLLECTION) .limit(1000) .find({ suppressAuth: true }) .then(function(result) { var jobs = result.items.map(function(item) { return { id: item._id || "", title: item.title || "", company: item.company || "", area: item.area || "", pref: item.pref || "", city: item.city || "", station: item.station || "", access: item.access || "", jobType: item.jobType || "", employment: item.employment || "", salaryText: item.salaryText || "", salaryMin: item.salaryMin || 0, salaryMax: item.salaryMax || 0, features: item.features || "", shift: item.shift || "", genre: item.genre || "", applyUrl: item.applyUrl || "" // ★ detail フィールドは省略（データ量削減のため） }; }); return ok({ headers: headers, body: JSON.stringify({ jobs: jobs, total: jobs.length }) }); }) .catch(function(err) { return serverError({ headers: headers, body: JSON.stringify({ error: String(err), jobs: [] }) }); }); } export function options_getJobs(request) { return ok({ headers: { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Methods": "GET, OPTIONS", "Access-Control-Allow-Headers": "Content-Type" }}); } // ──────────────────────────────────────────── // 2) GAS同期用：既存データ削除（POST） // ※ 何もせず200を返す（廃止） // ──────────────────────────────────────────── export function post_clearJobs(request) { return ok({ headers: { "Access-Control-Allow-Origin": "*", "Content-Type": "application/json" }, body: JSON.stringify({ ok: true, deleted: 0, message: "skipped" }) }); } export function options_clearJobs(request) { return ok({ headers: { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Methods": "POST, OPTIONS", "Access-Control-Allow-Headers": "Content-Type" }}); } // ──────────────────────────────────────────── // 3) GAS同期用：求人データ保存（POST） // jobIdをキーにsave（upsert） // ──────────────────────────────────────────── export function post_syncJobs(request) { var headers = { "Access-Control-Allow-Origin": "*", "Content-Type": "application/json; charset=utf-8" }; return request.body.json() .then(function(body) { var jobs = body.jobs || []; if (jobs.length === 0) { return serverError({ headers: headers, body: JSON.stringify({ error: "jobs is empty" }) }); } var promises = jobs.map(function(job) { var item = Object.assign({}, job); if (job.jobId) item._id = "job_" + job.jobId; return wixData.save(COLLECTION, item, { suppressAuth: true }); }); return Promise.all(promises) .then(function() { return ok({ headers: headers, body: JSON.stringify({ ok: true, count: jobs.length }) }); }); }) .catch(function(err) { return serverError({ headers: headers, body: JSON.stringify({ error: String(err) }) }); }); } export function options_syncJobs(request) { return ok({ headers: { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Methods": "POST, OPTIONS", "Access-Control-Allow-Headers": "Content-Type" }}); }
